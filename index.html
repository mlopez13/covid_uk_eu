Risk diagrams of COVID-19 in the UK and EU countries.¶

In this notebook, data from the European Comission's Joint Research Centre will be used to make risk diagrams of COVID-19 cumulative incidence and reproduction number.

It is impossible to know exactly how many people are infected in a pandemic, but the cumulative incidence $CI$ is an important factor, as it represents the new infections in the last $n$ days (in this case, 14). These are the people more prone to infecting other individuals, as they may not even know they have been infected yet. Crossing this value with the reproduction number $R$, which is roughly the ratio of new infections with respect to the original number of people with COVID-19, one obtains an estimate of the spread of the disease.

To be able to compare different countries, the $CI$ will be calculated as a proportion per 100,000 inhabitants. Therefore, the positive cases of infections will be divided by the total population of each country. This data is taken from the UN website: https://population.un.org/wpp/Download/Standard/Population/.

def preprocess(df):
    df = df.copy()
    
    # Select the countries defined in POPULATION.
    mask = df['CountryName'].isin(POPULATION.keys())
    df = df[mask]

    # Drop columns.
    drop_cols = ['iso3', 'lat', 'lon', 'EUcountry', 'EUCPMcountry', 'NUTS']
    df = df.drop(columns=drop_cols)

    # Rename columns and select the interesting ones.
    df.columns = ['DATE', 'COUNTRY', 'CUM_POS', 'CUM_DEC', 'CUM_REC', 'POS', 'HOS', 'INT']
    df = df[['DATE', 'COUNTRY', 'CUM_POS']]

    # Determine index and sort (no need to parse datetime, the format yyyy-mm-dd is good enough to sort).
    df = df.set_index('DATE')
    df = df.sort_index()
    
    return df

def process(df, countries):
    
    output = pd.DataFrame([])
    indexs = list(set(df.index.to_list()))
    
    for country in countries:
        
        # Subset of df.
        my_df = df.copy()
        mask = my_df['COUNTRY'] == country
        my_df = my_df[mask]

        # Ensure it is sorted.
        my_df = my_df.sort_index()
        
        # Compute Cumulative Incidence 14.
        my_df['POS'] = my_df['CUM_POS'] - my_df['CUM_POS'].shift(1)
        my_df['CUM14'] = my_df['POS'].rolling(window=14).sum()
        my_df['CI14'] = my_df['CUM14'] / POPULATION[country] * 1e5

        # Compute R7.
        pos = my_df['POS']
        my_df['R'] = (pos.shift(-1) + pos + pos.shift(1))/(pos.shift(4) + pos.shift(5) + pos.shift(6))

        r = my_df['R']
        my_df['R7'] = (r.shift(-3) + r.shift(-2) + r.shift(-1) + r + r.shift(1) + r.shift(2) + r.shift(3))/7

        # Drop rows with NAN values (due to computation with shifts).
        my_df = my_df.dropna()
        
        # Save result to new df.
        output = pd.concat([output, my_df])
        
        # Intersect dates.
        indexs = list(set(indexs) & set(my_df.index.to_list()))
    
    # Save only dates that all countries have and sort.
    output = output.loc[indexs]
    output = output.sort_index()
    
    return output

The risk ranges and the special calculation of $R$ are taken from the BIOCOMSC research group from the UPC: https://biocomsc.upc.edu/en.

def risk_plot(df, countries, ndays):
    
    df = df.copy()
    df = process(df, countries)
    
    # To determine the range needed to make the meshgrid...
    ci14 = []
    r7 = []
    
    for country in countries:
        my_df = df.copy()
        mask = my_df['COUNTRY'] == country
        my_df = my_df[mask].tail(ndays)
        
        ci14 = np.concatenate([ci14, my_df['CI14'].to_list()])
        r7 = np.concatenate([r7, my_df['R7'].to_list()])
    
    xmin = min(ci14)
    xmax = max(ci14)
    ymin = min(r7)
    ymax = max(r7)
    
    dx = (xmax - xmin)/4
    dy = (ymax - ymin)/4
    
    xmin = max(xmin - 3*dx, 0)
    xmax = xmax + dx
    ymin = max(ymin - 3*dy, 0)
    ymax = ymax + dy
    
    # Meshgrid.
    n = 100
    x = np.linspace(xmin, xmax, n)
    y = np.linspace(ymin, ymax, n)
    
    X, Y = np.meshgrid(x, y)
    
    # Risk = CI * R.
    Z = X*Y

    # For the title.
    date = my_df.index.values
    
    # PLOT.
    plt.figure(figsize=(10, 10))
    ax = plt.gca()
    ax.set_title('Risk diagram from {} to {}'.format(min(date), max(date)))
    ax.set_xlabel('CI$_{14}$ (cumulative incidence)')
    ax.set_ylabel('R$_7$ (reproduction number)')
    ax.set_xlim(xmin, xmax)
    ax.set_ylim(ymin, ymax)
    
    # Risk.
    im = ax.imshow(Z, origin='lower', interpolation='sinc', cmap=cmap,
                   extent=[xmin, xmax, ymin, ymax], aspect='auto', alpha=0.5)
    im.set_clim(0, 200)

    # Risk ranges.
    ax.contour(X, Y, Z, levels=[30, 200], colors='black', linewidths=1, linestyles='dashed')
    
    add_text(ax, xmin, xmax, ymin, ymax, 50, 0.1, 'Low risk')
    add_text(ax, xmin, xmax, ymin, ymax, 450, 0.2, 'Medium risk')
    add_text(ax, xmin, xmax, ymin, ymax, 1000, 0.3, 'High risk')
    
    # R = 1.
    ax.plot([0, xmax], [1, 1], color='black', linewidth=1, linestyle='dashed')
    
    # Scatter.
    for i, country in enumerate(countries):
        my_df = df.copy()
        mask = my_df['COUNTRY'] == country
        my_df = my_df[mask].tail(ndays)
        
        color = NAMED_COLORS[i%len(NAMED_COLORS)]
        marker = MARKERS[(i//len(NAMED_COLORS))%len(MARKERS)]
        
        ax.plot(my_df['CI14'], my_df['R7'], color=color)
        ax.scatter(my_df['CI14'], my_df['R7'], color=color, marker=marker, label=country)
        
        last = my_df.tail(1)
        
        ax.scatter(last['CI14'], last['R7'], facecolor='white', edgecolor='black', marker=marker, s=70)
    
    ax.legend()
    plt.show()
